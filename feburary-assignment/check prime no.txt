
#include <stdio.h>

int main() 
{
    int num;        
    printf("Enter the number\n");
    scanf("%d",&num);    

    int count=0;        
    
    for(  int i=2; i<num; i++)  
    {
        if( num%i == 0)
        count++;
    }
    if(count!=0)           
    {
        printf("not a prime number\n");
    }
    else
    {
        printf("prime number\n");
    }
    
    return 0;
}
